components/AlphaFixKeyer.tsx 
import React, { useRef, useState, useEffect } from 'react';

export default function AlphaFixKeyer() {
  const [low, setLow] = useState(30);
  const [high, setHigh] = useState(45);
  const [margin, setMargin] = useState(8);
  const [useTopo, setUseTopo] = useState(true);
  const [imageLoaded, setImageLoaded] = useState(false);
  const [logs, setLogs] = useState<string[]>(['Terminal ready. Drop an image to begin.']);

  const canvasRef = useRef<HTMLCanvasElement | null>(null);
  const sourceImgRef = useRef<HTMLImageElement | null>(null);

  const addLog = (msg: string) => {
    setLogs((prev) => [...prev, msg]);
  };

  const processImage = () => {
    const img = sourceImgRef.current;
    const canvas = canvasRef.current;
    if (!img || !canvas) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    canvas.width = img.naturalWidth;
    canvas.height = img.naturalHeight;
    ctx.drawImage(img, 0, 0);

    const w = canvas.width;
    const h = canvas.height;
    const imgData = ctx.getImageData(0, 0, w, h);
    const data = imgData.data;

    addLog(`Processing image: ${w}x${h}...`);

    // 1. Detect Checkerboard Grid Params in Corners
    const cornerW = Math.min(48, Math.floor(h / 8), Math.floor(w / 8));
    const samplePixels: { r: number; g: number; b: number; gray: number; x: number; y: number }[] = [];
    const checkMargin = 4;

    for (let y = checkMargin; y < cornerW; y++) {
      for (let x = checkMargin; x < cornerW; x++) {
        // Top-left corner
        const idx1 = (y * w + x) * 4;
        samplePixels.push({ r: data[idx1], g: data[idx1+1], b: data[idx1+2], gray: 0.299*data[idx1]+0.587*data[idx1+1]+0.114*data[idx1+2], x, y });
        // Top-right corner
        const x2 = w - 1 - x;
        const idx2 = (y * w + x2) * 4;
        samplePixels.push({ r: data[idx2], g: data[idx2+1], b: data[idx2+2], gray: 0.299*data[idx2]+0.587*data[idx2+1]+0.114*data[idx2+2], x: x2, y });
      }
    }

    if (samplePixels.length === 0) return;

    // Simple 1D K-Means to find light/dark checkerboard colors
    let c1 = 80, c2 = 180;
    for (let iter = 0; iter < 8; iter++) {
      let sum1 = 0, count1 = 0, sum2 = 0, count2 = 0;
      for (let p of samplePixels) {
        if (Math.abs(p.gray - c1) < Math.abs(p.gray - c2)) {
          sum1 += p.gray; count1++;
        } else {
          sum2 += p.gray; count2++;
        }
      }
      if (count1 > 0) c1 = sum1 / count1;
      if (count2 > 0) c2 = sum2 / count2;
    }
    const colorLow = Math.min(c1, c2);
    const colorHigh = Math.max(c1, c2);

    // Coarse-to-fine parameter fit to find size and offsets
    let bestScore = -1, bestSize = 16, bestOx = 0, bestOy = 0;
    for (let s = 8; s <= 48; s += 2) {
      for (let ox = 0; ox < s; ox += 4) {
        for (let oy = 0; oy < s; oy += 4) {
          let mae = 0;
          for (let p of samplePixels) {
            const phase = (Math.floor((p.x - ox) / s) + Math.floor((p.y - oy) / s)) % 2;
            const expected = phase === 0 ? colorLow : colorHigh;
            mae += Math.abs(p.gray - expected);
          }
          const score = 1.0 / (mae / samplePixels.length + 1e-5);
          if (score > bestScore) {
            bestScore = score; bestSize = s; bestOx = ox; bestOy = oy;
          }
        }
      }
    }

    addLog(`Grid size detected: ${bestSize}px, offsets: (${bestOx}, ${bestOy})`);

    // Sample dominant BGR colors
    let r1 = 0, g1 = 0, b1 = 0, count1 = 0;
    let r2 = 0, g2 = 0, b2 = 0, count2 = 0;
    for (let p of samplePixels) {
      const phase = (Math.floor((p.x - bestOx) / bestSize) + Math.floor((p.y - bestOy) / bestSize)) % 2;
      if (phase === 0) {
        r1 += p.r; g1 += p.g; b1 += p.b; count1++;
      } else {
        r2 += p.r; g2 += p.g; b2 += p.b; count2++;
      }
    }
    const rgb1 = [r1 / count1, g1 / count1, b1 / count1];
    const rgb2 = [r2 / count2, g2 / count2, b2 / count2];

    // Compute pixel-level distance to background colors
    const distMap = new Float32Array(w * h);
    for (let i = 0; i < w * h; i++) {
      const idx = i * 4;
      const r = data[idx], g = data[idx+1], b = data[idx+2];
      const dist1 = Math.sqrt((r-rgb1[0])**2 + (g-rgb1[1])**2 + (b-rgb1[2])**2);
      const dist2 = Math.sqrt((r-rgb2[0])**2 + (g-rgb2[1])**2 + (b-rgb2[2])**2);
      distMap[i] = Math.min(dist1, dist2);
    }

    // 2. Topological check via Breadth-First Search (BFS)
    const isBackground = new Uint8Array(w * h); // 1 = background, 0 = subject
    if (useTopo) {
      const visited = new Uint8Array(w * h);
      const queue: number[] = [];

      // Seed queue from edges
      for (let x = 0; x < w; x++) {
        queue.push(x); visited[x] = 1;
        const idxBottom = (h - 1) * w + x;
        queue.push(idxBottom); visited[idxBottom] = 1;
      }
      for (let y = 0; y < h; y++) {
        const idxLeft = y * w;
        if (!visited[idxLeft]) { queue.push(idxLeft); visited[idxLeft] = 1; }
        const idxRight = y * w + (w - 1);
        if (!visited[idxRight]) { queue.push(idxRight); visited[idxRight] = 1; }
      }

      let head = 0;
      while (head < queue.length) {
        const curr = queue[head++];
        const cx = curr % w;
        const cy = Math.floor(curr / w);

        // Distance matching background
        if (distMap[curr] <= high) {
          isBackground[curr] = 1;
          
          // Expand to neighbors
          const neighbors = [
            curr - 1, curr + 1, curr - w, curr + w
          ];
          for (let n of neighbors) {
            if (n >= 0 && n < w * h) {
              const nx = n % w;
              const ny = Math.floor(n / w);
              if (Math.abs(nx - cx) <= 1 && Math.abs(ny - cy) <= 1) { // bounds check
                if (!visited[n]) {
                  visited[n] = 1;
                  queue.push(n);
                }
              }
            }
          }
        }
      }
    } else {
      for (let i = 0; i < w * h; i++) {
        if (distMap[i] <= high) isBackground[i] = 1;
      }
    }

    // 3. Apply alpha matting and vignette crop
    for (let y = 0; y < h; y++) {
      for (let x = 0; x < w; x++) {
        const i = y * w + x;
        const idx = i * 4;

        if (x < margin || x >= w - margin || y < margin || y >= h - margin) {
          data[idx + 3] = 0; // Vignette crop
          continue;
        }

        if (isBackground[i] === 1) {
          const dist = distMap[i];
          const t = Math.max(0.0, Math.min(1.0, (dist - low) / (high - low)));
          const alphaVal = t * t * (3.0 - 2.0 * t);
          data[idx + 3] = Math.floor(alphaVal * 255);
        } else {
          data[idx + 3] = 255; // Solid subject
        }
      }
    }

    ctx.putImageData(imgData, 0, 0);
    addLog("Matting completed successfully!");
  };

  useEffect(() => {
    if (imageLoaded) {
      processImage();
    }
  }, [low, high, margin, useTopo, imageLoaded]);

  const handleDrop = (e: React.DragEvent) => {
    e.preventDefault();
    const file = e.dataTransfer.files[0];
    if (file && file.type.startsWith('image/')) {
      const reader = new FileReader();
      reader.onload = (event) => {
        if (sourceImgRef.current && event.target?.result) {
          sourceImgRef.current.src = event.target.result as string;
          sourceImgRef.current.onload = () => setImageLoaded(true);
        }
      };
      reader.readAsDataURL(file);
    }
  };

  return (
    <div className="min-h-screen bg-black text-white p-6">
      {/* Settings layout / Canvas components go here... */}
    </div>
  );
}
