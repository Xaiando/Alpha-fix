3. Mathematical Formulation

Let α(x) be the temporally stabilized matte (the output of your pass_15 EMA).
Let τ=0.5 be the geometric locus (contour pivot).
Let S(x) be the Signed Distance Field (SDF) relative to this contour, where S(x)>0 strictly inside the foreground (α≥τ) and S(x)<0 outside.

Let Er​(α)(x) and Dr​(α)(x) be the local OpenCV morphological erosion and dilation of α using a disk kernel of radius r.

We define a signed, spatially-variant advection velocity V(x)∈[−wmax​,wmax​]:
V(x)=wmax​⋅tanh(ρS(x)​)⋅exp(−2σ2S(x)2​)

From this, we extract two disjoint interpolation weights:
Wdilate​(x)=max(0,V(x))
Werode​(x)=max(0,−V(x))
Wstay​(x)=1.0−Wdilate​(x)−Werode​(x)

αnew​(x)=Wstay​(x)⋅α(x)+Werode​(x)⋅Er​(α)(x)+Wdilate​(x)⋅Dr​(α)(x)
4. The Core Mechanism

DCMT solves a level-set advection PDE (∂α/∂t=±∣∇α∣) but uses morphological supremum/infimum operations instead of finite differences, making it unconditionally stable and perfectly bounded to [0,1].

It dynamically blends the image with its own erosion or dilation depending on the physical distance from the pivot:

    Exterior (S<0): tanh is negative, Werode​ is active. Blending with the eroded matte forces the gradient to fetch lower alpha values from further out in the background. This geometrically drags the outer slope inward, crushing the tail footprint.

    Interior (S>0): tanh is positive, Wdilate​ is active. Blending with the dilated matte fetches higher alpha values from deeper inside. This pushes the interior gradient outward, fortifying the core against thinning.

    Isoline (S=0): V=0. αnew​=α. The exact geometric center of the transition never moves.

import cv2
import numpy as np

def apply_dcmt(alpha_ema, pivot=0.5, w_max=0.55, rho=1.5, sigma=3.0, radius=2):
    # 1. Target Isoline Mask
    m_fg = (alpha_ema >= pivot).astype(np.uint8)
    
    # 2. Signed Distance Field (S > 0 inside, S < 0 outside)
    d_in = cv2.distanceTransform(m_fg, cv2.DIST_L2, 3)
    d_out = cv2.distanceTransform(1 - m_fg, cv2.DIST_L2, 3)
    S = d_in - d_out
    
    # 3. Continuous Advection Weights
    tanh_term = np.tanh(S / rho)
    gauss_term = np.exp(-(S**2) / (2.0 * sigma**2))
    V = w_max * tanh_term * gauss_term
    
    W_dilate = np.clip(V, 0.0, 1.0)
    W_erode  = np.clip(-V, 0.0, 1.0)
    W_stay   = 1.0 - W_dilate - W_erode
    
    # 4. Morphological Fetch
    kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (2*radius + 1, 2*radius + 1))
    a_erode = cv2.erode(alpha_ema, kernel)
    a_dilate = cv2.dilate(alpha_ema, kernel)
    
    # 5. Guaranteed Convex Blend
    alpha_new = (W_stay * alpha_ema) + (W_erode * a_erode) + (W_dilate * a_dilate)
    
    return np.clip(alpha_new, 0.0, 1.0)

def apply_sclt(alpha_ema, pivot=0.5, k=0.4, sigma=2.0):
    m_fg = (alpha_ema >= pivot).astype(np.uint8)
    d_in = cv2.distanceTransform(m_fg, cv2.DIST_L2, 3)
    d_out = cv2.distanceTransform(1 - m_fg, cv2.DIST_L2, 3)
    S = d_out - d_in  # S > 0 OUTSIDE
    
    # Gradient of smoothed S
    S_smooth = cv2.GaussianBlur(S, (3, 3), 0)
    gx = cv2.Sobel(S_smooth, cv2.CV_32F, 1, 0, ksize=3)
    gy = cv2.Sobel(S_smooth, cv2.CV_32F, 0, 1, ksize=3)
    
    mag = np.sqrt(gx**2 + gy**2) + 1e-6
    nx, ny = gx / mag, gy / mag
    
    V = k * S * np.exp(-(S**2) / (2.0 * sigma**2))
    
    h, w = alpha_ema.shape
    y_idx, x_idx = np.indices((h, w), dtype=np.float32)
    
    map_x = x_idx + V * nx
    map_y = y_idx + V * ny
    
    alpha_new = cv2.remap(alpha_ema, map_x, map_y, cv2.INTER_LINEAR, borderMode=cv2.BORDER_REPLICATE)

import cv2
import numpy as np

def apply_osa_v3_htp(alpha_ema, alpha_0, pivot=0.5, kappa=2.5, R=2.0, sigma=2.0, omega=4.0, lam=5.0):
    m_fg = (alpha_ema >= pivot).astype(np.uint8)
    d_in = cv2.distanceTransform(m_fg, cv2.DIST_L2, 3)
    d_out = cv2.distanceTransform(1 - m_fg, cv2.DIST_L2, 3)
    S = d_out - d_in  # Repo convention: S > 0 is OUTSIDE
    
    S_smooth = cv2.GaussianBlur(S, (5, 5), 0)
    gx = cv2.Sobel(S_smooth, cv2.CV_32F, 1, 0, ksize=3)
    gy = cv2.Sobel(S_smooth, cv2.CV_32F, 0, 1, ksize=3)
    mag = np.sqrt(gx**2 + gy**2) + 1e-6
    nx, ny = gx / mag, gy / mag
    
    dist_beyond = np.maximum(0, np.abs(S) - R)
    E_plat = np.exp(-(dist_beyond**2) / (2.0 * sigma**2))
    W_lag = np.exp(-lam * np.abs(alpha_ema - alpha_0))
    
    V_rad = kappa * np.maximum(0, np.tanh(omega * (pivot - alpha_ema)))
    V_mag = np.clip(V_rad * E_plat * W_lag, 0.0, 0.85)
    
    h, w = alpha_ema.shape
    y_idx, x_idx = np.indices((h, w), dtype=np.float32)
    alpha_new = cv2.remap(alpha_ema, x_idx + V_mag * nx, y_idx + V_mag * ny, 
                          cv2.INTER_LINEAR, borderMode=cv2.BORDER_REPLICATE)
    return np.clip(alpha_new, 0.0, 1.0)

def apply_bar_lite(alpha_ema, alpha_0, pivot=0.5, gamma=3.0, beta=0.35, w_tilt=2.0, R=2.0, sigma=2.0, lam=5.0):
    m_fg = (alpha_ema >= pivot).astype(np.uint8)
    d_in = cv2.distanceTransform(m_fg, cv2.DIST_L2, 3)
    d_out = cv2.distanceTransform(1 - m_fg, cv2.DIST_L2, 3)
    S = d_out - d_in
    
    dist_beyond = np.maximum(0, np.abs(S) - R)
    E_plat = np.exp(-(dist_beyond**2) / (2.0 * sigma**2))
    W_lag = np.exp(-lam * np.abs(alpha_ema - alpha_0))
    
    U = pivot + beta * np.tanh(S / w_tilt)
    reaction = alpha_ema * (1.0 - alpha_ema) * (alpha_ema - U)
    
    alpha_new = alpha_ema + (gamma * reaction * E_plat * W_lag)
    return np.clip(alpha_new, 0.0, 1.0)

Python

    def _build_center_hole_alpha0(self, frame: np.ndarray, config) -> np.ndarray:
        h, w = frame.shape[:2]
        
        # Start with a solid white (opaque) alpha channel
        alpha0 = np.ones((h, w), dtype=np.float32)
        
        # OpenCV floodFill requires a mask padded by exactly 1 pixel on all sides
        ff_mask = np.zeros((h + 2, w + 2), dtype=np.uint8)
        
        # Seed point for the flood (dead center of the display capture hole)
        seed_pt = (w // 2, h // 2)
        
        # Tolerance: how much the AI noise can vary from the exact center pixel's color
        # 15 is a good starting point for compressed video
        lo_diff = getattr(config, 'center_hole_lo_diff', 15)
        up_diff = getattr(config, 'center_hole_up_diff', 15)
        
        # FIXED_RANGE means it compares every pixel to the original seed color,
        # preventing the flood from "creeping" up a gradient and escaping the window.
        flags = 4 | (255 << 8) | cv2.FLOODFILL_MASK_ONLY | cv2.FLOODFILL_FIXED_RANGE
        
        # Execute the flood fill. 
        # This draws a '255' in ff_mask ONLY where the connected center filler lives.
        cv2.floodFill(
            frame, 
            ff_mask, 
            seed_pt, 
            newVal=(0, 0, 0), # Ignored due to MASK_ONLY
            loDiff=(lo_diff, lo_diff, lo_diff), 
            upDiff=(up_diff, up_diff, up_diff), 
            flags=flags
        )
        
        # Crop the OpenCV padding back to the actual image dimensions
        hole_mask = ff_mask[1:-1, 1:-1]
        
        # Punch the hole into the alpha channel (0.0 = transparent)
        alpha0[hole_mask == 255] = 0.0
        
        return alpha0

    """
    h, w = frame.shape[:2]
    
    # 1. DEFINE THE THRESHOLDS (Wire these to config.py)
    # bg_thresh: Distance below this is 100% definitely the flat filler (The Spark)
    # fg_thresh: Distance above this is 100% definitely the character (The Wall)

    sure_fg = cv2.erode(sure_fg, kernel, iterations=3)

    unknown = cv2.subtract(np.ones((h, w), dtype=np.uint8) * 255, cv2.bitwise_or(sure_bg, sure_fg))
    
    # 3. SET THE MARKERS FOR THE PHYSICS ENGINE
    markers = np.zeros((h, w), dtype=np.int32)

    markers[sure_fg == 255] = 2 # 2 = Foreground (Armor)

def _apply_lgc_subject_crush(self, alpha_ema, frame, config):
    m_fg = (alpha_ema >= 0.5).astype(np.uint8)
    d_in = cv2.distanceTransform(m_fg, cv2.DIST_L2, 3)
    d_out = cv2.distanceTransform(1 - m_fg, cv2.DIST_L2, 3)
    S = d_out - d_in # S > 0 is OUTSIDE
    
    luma = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY).astype(np.float32) / 255.0
    
    # E_band: decays from 1.0 at boundary to 0.0 at radius
    E_band = np.clip(1.0 - (np.maximum(0, S) / config.lgc_radius), 0.0, 1.0)

1. The Core Upgrade: "The Pole of Inaccessibility"

If you want to automate floodFill, you cannot use the geometric centroid (center of mass) of the flat void. Many stream overlays have U-shaped or L-shaped chat windows. The geometric centroid of a "U" shape is in the dead space outside the hole entirely. If you drop a seed there, you will flood the decorative frame.

The Fix: Once we isolate a massive flat void, we run cv2.distanceTransform inside that specific void. The maximum value of this distance transform is mathematically guaranteed to be the pixel furthest away from any edge. It is the safest, deepest, most representative pixel in the entire hole. We drop the floodFill spark exactly there.

import cv2
import numpy as np

def _build_auto_hole_alpha0(self, frame: np.ndarray, config) -> np.ndarray:
    h, w = frame.shape[:2]
    
    # ---------------------------------------------------------
    # 1. STRUCTURAL SILENCE SCAN (Find the flatlands)
    # ---------------------------------------------------------
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    blur = cv2.GaussianBlur(gray, (5, 5), 0)
    
    gx = cv2.Sobel(blur, cv2.CV_32F, 1, 0, ksize=3)
    gy = cv2.Sobel(blur, cv2.CV_32F, 0, 1, ksize=3)
    grad_mag = np.sqrt(gx**2 + gy**2)
    
    # Flatness threshold (e.g., 6.0). Lower = stricter requirement for "flat" filler.
    flat_thresh = getattr(config, 'auto_hole_flat_thresh', 6.0)
    flat_mask = (grad_mag <= flat_thresh).astype(np.uint8)
    
    # Morphological open to destroy tiny flat islands (like smooth armor plates)
    kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
    flat_mask = cv2.morphologyEx(flat_mask, cv2.MORPH_OPEN, kernel)
    
    # ---------------------------------------------------------
    # 2. MASS FILTERING & DEEP SEEDING
    # ---------------------------------------------------------
    num_labels, labels, stats, _ = cv2.connectedComponentsWithStats(flat_mask, connectivity=8)
    
    # Only consider voids larger than X% of the screen (e.g., 1.5%)
    min_area = (h * w) * getattr(config, 'auto_hole_min_area_pct', 0.015)
    
    valid_seeds = []
    
    for i in range(1, num_labels): # Skip 0 (the noisy background/edges)
        if stats[i, cv2.CC_STAT_AREA] > min_area:
            # Isolate this specific massive hole
            comp_mask = (labels == i).astype(np.uint8)
            
            # Distance transform guarantees we find the deepest point inside the shape
            dist = cv2.distanceTransform(comp_mask, cv2.DIST_L2, 3)
            _, max_val, _, max_loc = cv2.minMaxLoc(dist)
            
            if max_val > 0:
                valid_seeds.append(max_loc) # max_loc is our safe (x, y) seed
                
    # ---------------------------------------------------------
    # 3. MULTI-SEED FLOOD FILL (The Radiometric Leash)
    # ---------------------------------------------------------
    if not valid_seeds:
        print("Warning: Auto-Hole found no voids. Falling back to center_hole.")
        # Assumes your _build_center_hole_alpha0 is available in the class
        return self._build_center_hole_alpha0(frame, config)
        
    alpha0 = np.ones((h, w), dtype=np.float32)
    ff_accum = np.zeros((h + 2, w + 2), dtype=np.uint8)
    
    # Piggyback on your proven center_hole tolerances
    lo_diff = getattr(config, 'center_hole_lo_diff', 15)
    up_diff = getattr(config, 'center_hole_up_diff', 15)
    flags = 4 | (255 << 8) | cv2.FLOODFILL_MASK_ONLY | cv2.FLOODFILL_FIXED_RANGE
    
    for seed_pt in valid_seeds:
        cv2.floodFill(
            frame, 
            ff_accum, 
            seed_pt, 
            newVal=(0, 0, 0), # Ignored for MASK_ONLY
            loDiff=(lo_diff, lo_diff, lo_diff), 
            upDiff=(up_diff, up_diff, up_diff), 
            flags=flags
        )
        
    # Punch the accumulated holes into the alpha channel
    hole_mask = ff_accum[1:-1, 1:-1]
    alpha0[hole_mask == 255] = 0.0
    
    return alpha0
